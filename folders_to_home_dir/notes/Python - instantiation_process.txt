#!/usr/bin/python3


class Point:
    # This method is created automatically
    def __new__(cls, *args, **kwargs):
        print("0. Create new instance of Point")
        return super().__new__(cls)

    def __init__(self, x, y):
        print("1. Initialize the new instance of Point")
        self.x = x
        self.y = y

    def __call__(self):
        print(f"2.The instance is called {self}")

    def __repr__(self) -> str:
        return f"{type(self).__name__}(x={self.x}, y={self.y})"


if __name__ == "__main__":
    point = Point(22, 45)
    print(point)
